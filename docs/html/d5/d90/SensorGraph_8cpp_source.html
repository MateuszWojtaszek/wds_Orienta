99void SensorGraph::addData(const QVector<int> &data) {
100 QChart *chartPtr = this->chart();
101 // Sprawdzenie warunków wstępnych: rozmiar danych, liczba serii, istnienie wykresu
102 if (data.size() != 3 || seriesList.size() != 3 || !chartPtr) return;
103
104 // Dodanie danych do każdej serii
105 for (int i = 0; i < 3; ++i) {
106 QLineSeries *series = seriesList.at(i); // Pobranie serii
107 if (!series) continue; // Pomiń, jeśli seria nie istnieje
108 series->append(currentSampleIndex, data[i]); // Dodanie punktu (indeks próbki, wartość)
109
110 // Usunięcie najstarszego punktu, jeśli przekroczono limit `sampleCount`
111 if (series->count() > sampleCount) {
112 series->remove(0); // Usuń pierwszy (najstarszy) punkt
113 }
114 }
115
116 // Okresowa aktualizacja zakresu osi X
117 const int xAxisUpdateFrequency = 10; // Częstotliwość aktualizacji osi X
118 bool isFirstSample = (currentSampleIndex == 1); // Sprawdzenie, czy to druga dodana próbka (indeks 1)
119
120 if (currentSampleIndex % xAxisUpdateFrequency == 0 || isFirstSample) {
121 if (auto *xAxis = qobject_cast<QValueAxis*>(chartPtr->axisX())) { // Bezpieczne rzutowanie na QValueAxis
122 qint64 lastIndex = currentSampleIndex; // Ostatni dodany indeks
123 // Obliczenie minimalnego indeksu X do wyświetlenia (przesuwne okno)
124 qint64 minX = (lastIndex >= sampleCount) ? (lastIndex - sampleCount + 1) : 0;
125 qint64 maxX = lastIndex; // Maksymalny indeks X to ostatni dodany
126
127 // Korekta zakresu, jeśli minX > maxX (może się zdarzyć przy małej liczbie próbek)
128 if (minX > maxX) {
129 minX = maxX > 0 ? maxX : 0;
130 }
131 // Korekta zakresu dla początkowych próbek, aby oś zaczynała się od 0
132 if (lastIndex < 1) {
133 minX = 0;
134 maxX = sampleCount > 0 ? sampleCount -1 : 0; // Pokaż pełen zakres, jeśli nie ma danych
135 }
136
137 xAxis->setRange(minX, maxX); // Ustawienie nowego zakresu osi X
138 }
139 }
140
141 currentSampleIndex++; // Inkrementacja indeksu próbki dla następnego wywołania
142}

154void SensorGraph::setSampleCount(int count) {
155 sampleCount = qMax(10, count); // Ustawienie nowej liczby próbek (nie mniej niż 10)
156 QChart *chartPtr = this->chart();
157
158 // Usunięcie nadmiarowych punktów z każdej serii
159 for (QLineSeries *series : seriesList) {
160 if (series) {
161 while (series->count() > sampleCount) {
162 series->remove(0); // Usuwanie najstarszych punktów
163 }
164 }
165 }
166
167 // Aktualizacja zakresu osi X
168 if (chartPtr) {
169 if (auto *xAxis = qobject_cast<QValueAxis*>(chartPtr->axisX())) {
170 qint64 minX = 0;
171 qint64 maxX = 0;
172
173 // Spróbuj ustalić zakres na podstawie istniejących punktów w pierwszej serii
174 if(!seriesList.isEmpty() && seriesList.first() && seriesList.first()->count() > 0){
175 const auto& points = seriesList.first()->points();
176 if (!points.isEmpty()) {
177 // Pobierz X pierwszego i ostatniego punktu
178 minX = static_cast<qint64>(points.first().x());
179 maxX = static_cast<qint64>(points.last().x());
180 }
181 } else {
182 // Jeśli brak punktów, ustaw domyślny zakres i zresetuj indeks próbki
183 maxX = (sampleCount > 0) ? (sampleCount - 1) : 0;
184 currentSampleIndex = 0; // Reset indeksu, jeśli czyścimy wykres przez zmianę sampleCount przy braku danych
185 }
186
187 // Ustaw zakres osi X, upewniając się, że minX <= maxX
188 if (maxX >= minX) {
189 xAxis->setRange(minX, maxX);
190 }
191 else {
192 // Fallback na domyślny zakres, jeśli obliczenia dały niespójny wynik
193 xAxis->setRange(0, sampleCount > 0 ? sampleCount - 1 : 0);
194 }
195 }
196 }
197}
