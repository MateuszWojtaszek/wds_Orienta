190bool MainWindow::loadSimulationData(const QString& pathToSimulationFile) { // Zmieniono nazwę parametru
191 QFile file(pathToSimulationFile);
192 if (file.open(QIODevice::ReadOnly | QIODevice::Text)) {
193 QTextStream in(&file);
194 loadedData.clear();
195 while (!in.atEnd()) {
196 QString line = in.readLine();
197 QStringList values = line.split(',', Qt::SkipEmptyParts);
198 if (values.size() == EXPECTED_DATA_SIZE_MW) {
199 QVector<float> dataToParse;
200 dataToParse.reserve(EXPECTED_DATA_SIZE_MW);
201 bool conversionOk = true;
202 for (const QString &val : values) {
203 float fVal = val.toFloat(&conversionOk);
204 if (!conversionOk) {
205 qWarning() << "Failed to convert value to float in line:" << line;
206 dataToParse.clear();
207 break;
208 }
209 dataToParse.append(fVal);
210 }
211 if (!dataToParse.isEmpty()) {
212 loadedData.append(dataToParse);
213 }
214 } else if (!line.trimmed().isEmpty()) {
215 qWarning() << "Skipping line with incorrect number of values (" << values.size() << "):" << line;
216 }
217 }
218 file.close();
219 qInfo() << "Successfully loaded" << loadedData.size() << "lines of simulation data from" << pathToSimulationFile;
220 return true;
221 } else {
222 qWarning() << "Failed to open simulation data file:" << pathToSimulationFile << "Error:" << file.errorString();
223 return false;
224 }
225}

227void MainWindow::processImuData(const QVector<float>& acquiredData) {
228 if (acquiredData.size() != EXPECTED_DATA_SIZE_MW) {
229 qWarning() << "Received data packet with incorrect size:" << acquiredData.size() << "Expected:" << EXPECTED_DATA_SIZE_MW;
230 return;
231 }
232
233 QVector<int> gyro = { static_cast<int>(acquiredData[GYRO_X_IDX_MW]), static_cast<int>(acquiredData[GYRO_Y_IDX_MW]), static_cast<int>(acquiredData[GYRO_Z_IDX_MW]) };
234 QVector<int> acc = { static_cast<int>(acquiredData[ACC_X_IDX_MW]), static_cast<int>(acquiredData[ACC_Y_IDX_MW]), static_cast<int>(acquiredData[ACC_Z_IDX_MW]) };
235 QVector<int> mag = { static_cast<int>(acquiredData[MAG_X_IDX_MW]), static_cast<int>(acquiredData[MAG_Y_IDX_MW]), static_cast<int>(acquiredData[MAG_Z_IDX_MW]) };
236
237 float roll = acquiredData[ROLL_IDX_MW];
238 float pitch = acquiredData[PITCH_IDX_MW];
239 float yaw = acquiredData[YAW_IDX_MW];
240
241 if (imuHandler) {
242 imuHandler->updateData(acc, gyro, mag);
243 imuHandler->setRotation(yaw, pitch, roll);
244
245 // Użyj std::abs do porównywania floatów z zerem, aby uniknąć problemów z precyzją
246 if (std::abs(acquiredData[MAG_X_IDX_MW]) > 1e-6 || std::abs(acquiredData[MAG_Y_IDX_MW]) > 1e-6) {
247 float heading_rad = std::atan2(acquiredData[MAG_Y_IDX_MW], acquiredData[MAG_X_IDX_MW]);
248 float heading_deg = heading_rad * 180.0f / static_cast<float>(M_PI); // Użyj M_PI z <cmath>
249 if (heading_deg < 0.0f) {
250 heading_deg += 360.0f;
251 }
252 imuHandler->updateCompass(heading_deg);
253 } else {
254 imuHandler->updateCompass(0.0f);
255 }
256 } else {
257 qWarning() << "processImuData called but imuHandler is null!";
258 }
259}

313void MainWindow::updateSimulatedGPSMarker() {
314 if (simulationMode && gpsHandler) {
315 double angleRad = static_cast<double>(currentDataIndex) * GPS_OSCILLATION_SPEED_FACTOR_MW;
316 double latOffset = GPS_OSCILLATION_AMPLITUDE_MW * std::sin(angleRad);
317 double lonOffset = GPS_OSCILLATION_AMPLITUDE_MW * std::cos(angleRad);
318
319 double currentLatitude = BASE_LATITUDE_MW + latOffset;
320 double currentLongitude = BASE_LONGITUDE_MW + lonOffset;
321
322 gpsHandler->updateMarker(currentLatitude, currentLongitude);
323 }

327void MainWindow::toggleSimulationMode() {
328 simulationMode = !simulationMode;
329 currentDataIndex = 0; // Zresetuj indeks danych przy każdej zmianie trybu
330
331 QAction* simAction = nullptr;
332 QList<QAction*> actions = menuBar()->findChildren<QAction*>();
333 for(QAction* action : actions) {
334 if(action->objectName() == "simulationModeAction") {
335 simAction = action;
336 break;
337 }
338 }
339
340 if (simulationMode) {
341 if (serialConnected) { // Jeśli port jest połączony, zamknij go
342 serialHandler->closePort();
343 serialConnected = false;
344 qInfo() << "Serial port closed due to enabling simulation mode.";
345 }
346 if (!loadedData.isEmpty()) {
347 if (gpsHandler) { // Zaktualizuj pozycję GPS na startową
348 gpsHandler->updateMarker(BASE_LATITUDE_MW, BASE_LONGITUDE_MW);
349 }
350 simulationTimer->start(SIMULATION_TIMER_INTERVAL_MS_MW);
351 qInfo() << "Simulation mode enabled. Timer started.";
352 } else {
353 QMessageBox::warning(this, tr("Simulation Mode"), tr("Simulation mode enabled, but no simulation data loaded."));
354 qInfo() << "Simulation mode enabled, but no data loaded. Disabling simulation mode.";
355 simulationMode = false; // Cofnij zmianę, jeśli nie ma danych
356 }
357 } else { // Wyłączanie trybu symulacji
358 simulationTimer->stop();
359 qInfo() << "Simulation mode disabled. Timer stopped.";
360 }
361
362 if (simAction) { // Zaktualizuj stan zaznaczenia w menu
363 simAction->setChecked(simulationMode);
364 }
365}

367void MainWindow::selectPort() {
368 QStringList ports;
369 const auto availablePorts = QSerialPortInfo::availablePorts();
370 for (const QSerialPortInfo &info : availablePorts) {
371 ports << info.portName();
372 }
373
374 if (ports.isEmpty()) {

378

401void MainWindow::updateSimulationData() {
402 if (serialConnected || !simulationMode || loadedData.isEmpty()) {
403 simulationTimer->stop(); // Zatrzymaj, jeśli nie jesteśmy w trybie symulacji lub nie ma danych

412 processImuData(currentFrameData);
413 updateSimulatedGPSMarker();
