146bool MainWindow::loadSimulationData(const QString& filePath) {
147 QFile file(filePath); // Utworzenie obiektu QFile
148 // Próba otwarcia pliku do odczytu w trybie tekstowym
149 if (file.open(QIODevice::ReadOnly | QIODevice::Text)) {
150 QTextStream in(&file); // Strumień tekstowy do odczytu z pliku
151 loadedData.clear(); // Wyczyszczenie ewentualnych starych danych
152 // Odczyt pliku linia po linii
153 while (!in.atEnd()) {
154 QString line = in.readLine(); // Odczytanie linii
155 // Podział linii na części względem przecinka, pomijając puste części
156 QStringList values = line.split(',', Qt::SkipEmptyParts);
157 // Sprawdzenie, czy liczba wartości zgadza się z oczekiwaną
158 if (values.size() == EXPECTED_DATA_SIZE) {
159 QVector<float> parsedLine; // Wektor na sparsowane wartości z linii
160 parsedLine.reserve(EXPECTED_DATA_SIZE); // Rezerwacja miejsca dla optymalizacji
161 bool conversionOk = true; // Flaga poprawności konwersji
162 // Iteracja przez wartości tekstowe
163 for (const QString &val : values) {
164 float fVal = val.toFloat(&conversionOk); // Konwersja na float
165 // Sprawdzenie, czy konwersja się powiodła
166 if (!conversionOk) {
167 qWarning() << "Failed to convert value to float in line:" << line; // Logowanie błędu
168 parsedLine.clear(); // Wyczyść częściowo sparsowaną linię
169 break; // Przerwij przetwarzanie tej linii
170 }
171 parsedLine.append(fVal); // Dodanie sparsowanej wartości do wektora
172 }
173 // Jeśli linia została poprawnie sparsowana (nie była pusta po błędzie konwersji)
174 if (!parsedLine.isEmpty()) {
175 loadedData.append(parsedLine); // Dodanie sparsowanej linii do głównych danych
176 }
177 } else if (!line.trimmed().isEmpty()) { // Jeśli linia nie jest pusta, ale ma złą liczbę wartości
178 qWarning() << "Skipping line with incorrect number of values (" << values.size() << "):" << line; // Logowanie ostrzeżenia
179 }
180 }
181 file.close(); // Zamknięcie pliku
182 qInfo() << "Successfully loaded" << loadedData.size() << "lines of simulation data from" << filePath; // Logowanie sukcesu
183 return true; // Zwrócenie informacji o sukcesie
184 } else {
185 // Logowanie błędu, jeśli nie udało się otworzyć pliku
186 qWarning() << "Failed to open simulation data file:" << filePath << "Error:" << file.errorString();
187 return false; // Zwrócenie informacji o porażce
188 }

201void MainWindow::toggleSimulationMode() {
202 simulationMode = !simulationMode; // Przełączenie flagi trybu symulacji
203 currentDataIndex = 0; // Zresetowanie indeksu danych symulacyjnych
204
205 if (simulationMode) { // Włączanie trybu symulacji
206 // Jeśli port szeregowy jest połączony, zamknij go
207 if (serialConnected) {
208 serialHandler->closePort();
209 serialConnected = false;
210 qInfo() << "Serial port closed due to enabling simulation mode.";
211 }
212 // Sprawdź, czy dane symulacyjne są załadowane
213 if (!loadedData.isEmpty()) {
214 // Aktualizacja pozycji startowej markera GPS (jeśli handler istnieje)
215 if (gpsHandler) {
216 gpsHandler->updateMarker(baseLatitude, baseLongitude);
217 }
218 simulationTimer->start(SIMULATION_TIMER_INTERVAL_MS); // Uruchomienie timera symulacji
219 qInfo() << "Simulation mode enabled. Timer started.";
220 } else {
221 // Ostrzeżenie, jeśli próbowano włączyć symulację bez danych
222 QMessageBox::warning(this, tr("Simulation Mode"), tr("Simulation mode enabled, but no simulation data loaded."));
223 qInfo() << "Simulation mode enabled, but no data loaded.";
224 simulationMode = false; // Cofnięcie zmiany trybu
225 // Aktualizacja stanu przycisku menu - odznaczenie
226 QList<QAction*> actions = menuBar()->findChildren<QAction*>();
227 for(QAction* action : actions) {
228 if(action->text() == tr("Simulation Mode")) {
229 action->setChecked(false); // Odznaczenie akcji w menu
230 break;
231 }
232 }
233 }
234 } else { // Wyłączanie trybu symulacji
235 simulationTimer->stop(); // Zatrzymanie timera
236 qInfo() << "Simulation mode disabled. Timer stopped.";
237 // Aktualizacja stanu przycisku menu - odznaczenie
238 QList<QAction*> actions = menuBar()->findChildren<QAction*>();
239 for(QAction* action : actions) {
240 if(action->text() == tr("Simulation Mode")) {
241 action->setChecked(simulationMode); // Ustawienie stanu akcji (na false)
242 break;
243 }
244 }
245 }
246 // Ponowna (lub pierwsza, jeśli nie było danych) aktualizacja stanu przycisku menu
247 QList<QAction*> actions = menuBar()->findChildren<QAction*>();
248 for(QAction* action : actions) {
249 if(action->text() == tr("Simulation Mode")) {
250 action->setChecked(simulationMode); // Ustawienie właściwego stanu akcji
251 break;
252 }
253 }

266void MainWindow::updateSimulationData() {
267 // Warunki przerwania symulacji: połączenie szeregowe, wyłączony tryb symulacji, brak danych
268 if (serialConnected || !simulationMode || loadedData.isEmpty()) {
269 simulationTimer->stop(); // Zatrzymaj timer, jeśli warunki są spełnione
270 return; // Zakończ funkcję
271 }
272
273 // Sprawdź, czy osiągnięto koniec danych symulacyjnych
274 if (checkSimulationEndAndUpdateState()) {
275 return; // Jeśli symulacja się zakończyła, zakończ funkcję
276 }
277
278 // Pobranie bieżącego zestawu danych IMU z załadowanych danych
279 const QVector<float> &imuData = loadedData[currentDataIndex];
280 processImuData(imuData); // Przetworzenie danych IMU
281
282 updateSimulatedGPSMarker(); // Aktualizacja symulowanej pozycji GPS
283
284 currentDataIndex++; // Przejście do następnego zestawu danych
285}

299 // Sprawdzenie, czy indeks wyszedł poza zakres załadowanych danych

307 QList<QAction*> actions = menuBar()->findChildren<QAction*>();
308 for(QAction* action : actions) {
309 if(action->text() == tr("Simulation Mode")) {

312 }
313 }
314 return true; // Zwróć true, informując, że symulacja się zakończyła

316 return false; // Zwróć false, symulacja trwa nadal

330void MainWindow::updateSimulatedGPSMarker() {
331 // Sprawdź, czy tryb symulacji jest aktywny i czy handler GPS istnieje
332 if (simulationMode && gpsHandler) {
333 // Obliczenie kąta oscylacji na podstawie indeksu danych i współczynnika prędkości
334 double angleRad = static_cast<double>(currentDataIndex) * gpsOscillationSpeedFactor;
335 // Obliczenie przesunięć dla szerokości i długości geograficznej
336 double latOffset = gpsOscillationAmplitude * std::sin(angleRad);
337 double lonOffset = gpsOscillationAmplitude * std::cos(angleRad);
338
339 // Obliczenie aktualnych współrzędnych GPS
340 double currentLatitude = baseLatitude + latOffset;
341 double currentLongitude = baseLongitude + lonOffset;
342
343 // Aktualizacja pozycji markera w handlerze GPS
344 gpsHandler->updateMarker(currentLatitude, currentLongitude);
345 }

379void MainWindow::processImuData(const QVector<float>& data) {
380 // Sprawdzenie, czy rozmiar otrzymanych danych jest zgodny z oczekiwanym
381 if (data.size() != EXPECTED_DATA_SIZE) {
382 qWarning() << "Received data packet with incorrect size:" << data.size() << "Expected:" << EXPECTED_DATA_SIZE;
383 return; // Ignoruj niepoprawny pakiet danych
384 }
385
386 // Ekstrakcja danych żyroskopu (konwersja na int - może wymagać weryfikacji, czy to zamierzone)
387 QVector<int> gyro = { static_cast<int>(data[GYRO_X_IDX]), static_cast<int>(data[GYRO_Y_IDX]), static_cast<int>(data[GYRO_Z_IDX]) };
388 // Ekstrakcja danych akcelerometru (konwersja na int)
389 QVector<int> acc = { static_cast<int>(data[ACC_X_IDX]), static_cast<int>(data[ACC_Y_IDX]), static_cast<int>(data[ACC_Z_IDX]) };
390 // Ekstrakcja danych magnetometru (konwersja na int)
391 QVector<int> mag = { static_cast<int>(data[MAG_X_IDX]), static_cast<int>(data[MAG_Y_IDX]), static_cast<int>(data[MAG_Z_IDX]) };
392
393 // Ekstrakcja kątów orientacji
394 float roll = data[ROLL_IDX];
395 float pitch = data[PITCH_IDX];
396 float yaw = data[YAW_IDX];
397
398 // Sprawdzenie, czy handler IMU istnieje
399 if (imuHandler) {
400 // Aktualizacja danych surowych w handlerze IMU
401 imuHandler->updateData(acc, gyro, mag);
402 // Ustawienie rotacji modelu 3D w handlerze IMU
403 imuHandler->setRotation(yaw, pitch, roll);
404
405 // Obliczenie i aktualizacja kursu kompasu
406 // Sprawdzenie, czy dane magnetometru są wystarczające i niezerowe (aby uniknąć atan2(0,0))
407 if (mag.size() >= 2 && (mag[0] != 0 || mag[1] != 0)) {
408 // Obliczenie kąta za pomocą atan2 (daje wynik w radianach od -PI do PI)
409 float heading_rad = std::atan2(static_cast<float>(mag[1]), static_cast<float>(mag[0]));
410 // Konwersja radianów na stopnie
411 float heading_deg = heading_rad * 180.0f / M_PI;
412 // Normalizacja kąta do zakresu [0, 360) stopni
413 if (heading_deg < 0) {
414 heading_deg += 360.0f;
415 }
416 // Aktualizacja wskaźnika kompasu w handlerze IMU
417 imuHandler->updateCompass(heading_deg);
418 } else {
419 // Jeśli dane magnetometru są niewystarczające lub zerowe, ustaw kurs na 0
420 imuHandler->updateCompass(0.0f);
421 }
422
423 } else {
424 // Logowanie ostrzeżenia, jeśli handler IMU nie został zainicjalizowany
425 qWarning() << "processImuData called but imuHandler is null!";
426 }
427}

467void MainWindow::selectPort() {
468 QStringList ports; // Lista nazw portów
469 // Pobranie informacji o dostępnych portach szeregowych
470 const auto availablePorts = QSerialPortInfo::availablePorts();
471 for (const QSerialPortInfo &info : availablePorts) {
472 ports << info.portName(); // Dodanie nazwy portu do listy
473 }
474
475 // Sprawdzenie, czy znaleziono jakiekolwiek porty
476 if (ports.isEmpty()) {
477 // Wyświetlenie ostrzeżenia, jeśli brak dostępnych portów
478 QMessageBox::warning(this, tr("No Ports"), tr("No serial ports available. Check connections and permissions."));
479 return; // Zakończenie funkcji
480 }
481
482 bool ok; // Flaga wskazująca, czy użytkownik kliknął OK w dialogu
483 // Wyświetlenie dialogu QInputDialog typu lista rozwijana
484 QString port = QInputDialog::getItem(this, tr("Select Port"), tr("Available Ports:"), ports, 0, false, &ok);
485
486 // Sprawdzenie, czy użytkownik wybrał port i kliknął OK
487 if (ok && !port.isEmpty()) {
488 handlePortConnectionAttempt(port); // Próba nawiązania połączenia z wybranym portem
489 }

502void MainWindow::handlePortConnectionAttempt(const QString& portName) {
503 selectedPort = portName; // Zapisanie nazwy wybranego portu
504
505 // Jeśli już istnieje połączenie, zamknij je przed próbą otwarcia nowego
506 if (serialConnected) {
507 serialHandler->closePort();
508 serialConnected = false;
509 }
510
511 // Jeśli tryb symulacji jest aktywny, wyłącz go
512 if (simulationMode) {
513 simulationTimer->stop(); // Zatrzymaj timer
514 simulationMode = false; // Zmień flagę
515 qInfo() << "Simulation mode disabled due to serial port selection.";
516 // Odznacz akcję "Simulation Mode" w menu
517 QList<QAction*> actions = menuBar()->findChildren<QAction*>();
518 for(QAction* action : actions) {
519 if(action->text() == tr("Simulation Mode")) {
520 action->setChecked(false);
521 break;
522 }
523 }
524 }
525
526 // Próba otwarcia wybranego portu za pomocą handlera
527 if (serialHandler->openPort(selectedPort)) {
528 serialConnected = true; // Ustawienie flagi połączenia na true
529 // Wyświetlenie komunikatu o sukcesie
530 QMessageBox::information(this, tr("Connected"), tr("Successfully connected to %1").arg(portName));
531 qInfo() << "Connected to serial port:" << selectedPort; // Logowanie sukcesu
532 } else {
533 // Wyświetlenie komunikatu o błędzie połączenia
534 QMessageBox::critical(this, tr("Connection Error"), tr("Failed to open port %1. Reason: %2").arg(portName).arg(serialHandler->getLastError()));
535 qWarning() << "Failed to open serial port:" << selectedPort; // Logowanie błędu
536 serialConnected = false; // Upewnienie się, że flaga połączenia jest false
537 }
