269QWidget* ImuDataHandler::createBarDisplayWidget() {
270 QWidget *barWidget = new QWidget();
271 QVBoxLayout *barLayout = new QVBoxLayout(barWidget);
272
273 // Funkcja pomocnicza do tworzenia grupy pasków dla jednego sensora
274 auto addBarGroup = [&](const QString &title, QProgressBar *&x, QProgressBar *&y, QProgressBar *&z, int range) {
275 QGroupBox* groupBox = new QGroupBox(title); // Grupa dla sensora
276 QVBoxLayout* groupLayout = new QVBoxLayout(groupBox); // Layout wewnątrz grupy
277
278 // Funkcja pomocnicza do tworzenia pojedynczego paska (X, Y lub Z)
279 auto createBar = [&](const QString &axis, QProgressBar *&bar) {
280 QLabel *axisLabel = new QLabel(axis); // Etykieta osi (X:, Y:, Z:)
281 axisLabel->setFixedWidth(20); // Stała szerokość etykiety
282 bar = new QProgressBar(); // Tworzenie paska postępu
283 bar->setRange(-range, range); // Ustawienie zakresu
284 bar->setValue(0); // Wartość początkowa
285 bar->setTextVisible(true); // Wyświetlanie tekstu na pasku
286 // Format wyświetlania tekstu (początkowo pokazujący zakres)
287 bar->setFormat(QString("%1 / %2").arg(QString::number(range)).arg(QString::number(-range)));
288
289 // Dodanie etykiet z zakresem (-range, 0, range) pod paskiem
290 QHBoxLayout *rangeLabels = new QHBoxLayout();
291 rangeLabels->addWidget(new QLabel(QString::number(-range)));
292 rangeLabels->addStretch(); // Rozciągliwa przestrzeń
293 rangeLabels->addWidget(new QLabel("0"));
294 rangeLabels->addStretch(); // Rozciągliwa przestrzeń
295 rangeLabels->addWidget(new QLabel(QString::number(range)));
296
297 // Layout pionowy dla paska i etykiet zakresu
298 QVBoxLayout *barWithLabels = new QVBoxLayout();
299 barWithLabels->addLayout(rangeLabels);
300 barWithLabels->addWidget(bar);
301 barWithLabels->setSpacing(0); // Brak odstępu między etykietami a paskiem
302
303 // Layout poziomy dla etykiety osi i paska z etykietami zakresu
304 QHBoxLayout *row = new QHBoxLayout();
305 row->addWidget(axisLabel);
306 row->addLayout(barWithLabels);
307 groupLayout->addLayout(row); // Dodanie wiersza do layoutu grupy
308 };
309
310 // Utworzenie pasków dla osi X, Y, Z
311 createBar("X:", x);
312 createBar("Y:", y);
313 createBar("Z:", z);
314 barLayout->addWidget(groupBox); // Dodanie grupy do głównego layoutu pasków
315 };
316
317 // Utworzenie grup pasków dla każdego sensora
318 addBarGroup("Accelerometer [mg]", accXBar, accYBar, accZBar, 4000);
319 addBarGroup("Gyroscope [dps]", gyroXBar, gyroYBar, gyroZBar, 250);
320 addBarGroup("Magnetometer [mG]", magXBar, magYBar, magZBar, 1600);
321
322 return barWidget;
323}

463void ImuDataHandler::setupModelLoader3D(Qt3DCore::QEntity* rootEntity) {
464 // Tworzenie bytu, który będzie przechowywał ładowany model
465 auto *modelEntity = new Qt3DCore::QEntity(rootEntity);
466 // Tworzenie komponentu do ładowania sceny 3D
467 auto *loader = new Qt3DRender::QSceneLoader(modelEntity);
468
469 // Połączenie sygnału zmiany statusu ładowarki
470 connect(loader, &Qt3DRender::QSceneLoader::statusChanged, this,
471 [this, loader, modelEntity](Qt3DRender::QSceneLoader::Status status) {
472 // Sprawdzenie, czy ładowanie zakończyło się sukcesem (Ready)
473 if (status == Qt3DRender::QSceneLoader::Ready) {
474 // Pobranie listy bytów załadowanych przez loader
475 const auto &childEntities = loader->entities();
476 if (!childEntities.isEmpty()) {
477 // Pobranie pierwszego (zazwyczaj głównego) bytu załadowanej sceny
478 Qt3DCore::QEntity *sceneRootEntity = childEntities.first();
479 if (!sceneRootEntity) return; // Zabezpieczenie, jeśli byt jest nullem
480
481 // Utworzenie transformacji dla płytki, jeśli jeszcze nie istnieje
482 if (!this->boardTransform) {
483 this->boardTransform = new Qt3DCore::QTransform();
484 qDebug() << "BoardTransform created.";
485 }
486
487 // Sprawdzenie, czy transformacja nie jest już dodana (na wszelki wypadek)
488 bool alreadyHasTransform = false;
489 for(Qt3DCore::QComponent *comp : sceneRootEntity->components()) {
490 if(comp == this->boardTransform) {
491 alreadyHasTransform = true;
492 break;
493 }
494 }
495 // Dodanie transformacji do głównego bytu załadowanej sceny, jeśli jeszcze jej nie ma
496 if(!alreadyHasTransform) {
497 qDebug() << "Attaching boardTransform.";
498 sceneRootEntity->addComponent(this->boardTransform);
499 }
500 } else {
501 qWarning() << "SceneLoader loaded successfully, but no entities found.";
502 }
503 // Sprawdzenie, czy wystąpił błąd podczas ładowania
504 } else if (status == Qt3DRender::QSceneLoader::Error) {
505 qWarning() << "Failed to load model (status Error). Source:" << loader->source();
506 }
507 });
508
509 // Ustawienie źródła modelu 3D (ścieżka do pliku)
510 // WAŻNE: Upewnij się, że ta ścieżka jest poprawna w Twoim systemie!
511 loader->setSource(QUrl::fromLocalFile("/Users/mateuszwojtaszek/projekty/wds_Orienta/ESP32.dae"));
512 // Dodanie komponentu loadera do bytu modelu
513 modelEntity->addComponent(loader);
514}

626Qt3DCore::QEntity* ImuDataHandler::createCompassArrow(Qt3DCore::QEntity* parentEntity) {
627 // Tworzenie bytu dla strzałki
628 auto *arrowEntity = new Qt3DCore::QEntity(parentEntity);
629 // Tworzenie siatki w kształcie prostopadłościanu (wydłużonego, aby przypominał strzałkę)
630 auto *cuboidMesh = new Qt3DExtras::QCuboidMesh();
631 cuboidMesh->setXExtent(1.0f); // Szerokość
