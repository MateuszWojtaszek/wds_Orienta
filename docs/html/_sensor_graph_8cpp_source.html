23SensorGraph::SensorGraph(const QString &title, int minY, int maxY, QWidget *parent)
24 : QChartView(new QChart(), parent), // Inicjalizacja bazowego QChartView z nowym QChart
25 defaultSampleCount(1000), // Inicjalizacja domyślnej liczby próbek
26 currentSampleIndex(0) // Inicjalizacja indeksu próbki
27{
28 // Pobranie wskaźnika na zarządzany obiekt QChart
29 QChart *chartPtr = this->chart();
30 if (!chartPtr) {
31 // Podstawowa obsługa błędu, jeśli chart() zwróci nullptr
32 qWarning() << "SensorGraph: Failed to get chart object!";
33 return;
34 }
35 chartPtr->setTitle(title); // Ustawienie tytułu wykresu
36
37 // Konfiguracja legendy
38 QLegend *legend = chartPtr->legend();
39 if (legend) {
40 legend->setVisible(true);
41 legend->setAlignment(Qt::AlignTop); // Wyrównanie do góry
42 legend->setMarkerShape(QLegend::MarkerShapeCircle); // Kształt znacznika w legendzie
43 }
44
45 // Nazwy i kolory dla serii danych (X, Y, Z)
46 QStringList axisNames = {"X", "Y", "Z"};
47 QList<QColor> colors = {Qt::blue, Qt::red, Qt::green};
48
49 // Utworzenie i konfiguracja trzech serii danych
50 for (int i = 0; i < 3; ++i) {
51 auto *series = new QLineSeries(); // Utworzenie nowej serii liniowej
52 series->setName(axisNames[i]); // Ustawienie nazwy widocznej w legendzie
53 series->setColor(colors[i]); // Ustawienie koloru linii
54 chartPtr->addSeries(series); // Dodanie serii do wykresu
55 seriesList.append(series); // Dodanie wskaźnika serii do listy zarządzającej
56 }
57
58 // Utworzenie i konfiguracja osi X (indeks próbki/czas)
59 auto xAxis = new QValueAxis();
60 xAxis->setTitleText("Sample Index"); // Etykieta osi X
61 xAxis->setTickCount(11); // Sugerowana liczba głównych znaczników na osi
62 xAxis->setLabelFormat("%lld"); // Formatowanie etykiet (dla qint64)
63 chartPtr->setAxisX(xAxis); // Ustawienie osi X dla wykresu
64
65 // Utworzenie i konfiguracja osi Y (wartość)
66 auto yAxis = new QValueAxis();
67 yAxis->setTitleText("Value"); // Etykieta osi Y
68 yAxis->setRange(minY, maxY); // Ustawienie początkowego zakresu osi Y
69 chartPtr->setAxisY(yAxis); // Ustawienie osi Y dla wykresu
70
71 // Powiązanie każdej serii z osiami X i Y
72 // W Qt 6 można by użyć chartPtr->setAxisX(xAxis, seriesList); itp.
73 // ale dla kompatybilności z Qt 5 lepiej przypisać indywidualnie.
74 for (QLineSeries *series : seriesList) {
75 if (series) {
76 chartPtr->setAxisX(xAxis, series);
77 chartPtr->setAxisY(yAxis, series);
78 }
79 }
80
81 // Ustawienie początkowego zakresu osi X
82 // Zakres od 0 do defaultSampleCount-1 (lub 0 jeśli defaultSampleCount <= 0)
83 xAxis->setRange(0, defaultSampleCount > 0 ? defaultSampleCount - 1 : 0);
84
85 // Ustawienie polityki rozmiaru, aby widget mógł się rozciągać
86 this->setSizePolicy(QSizePolicy::Expanding, QSizePolicy::MinimumExpanding);
87 this->setRenderHint(QPainter::Antialiasing); // Włączenie antyaliasingu dla gładszych linii
88}

91void SensorGraph::addData(const QVector<int> &axisValuesToAdd) {
92 QChart *chartPtr = this->chart();
93 // Sprawdzenie warunków wstępnych: poprawny rozmiar wektora, liczba serii, istnienie wykresu
94 if (axisValuesToAdd.size() != 3 || seriesList.size() != 3 || !chartPtr) {
95 // Można dodać qWarning() w razie potrzeby
96 return;
97 }
98
99 // Dodanie nowych punktów do każdej z trzech serii
100 for (int i = 0; i < 3; ++i) {
101 QLineSeries *series = seriesList.at(i);
102 if (!series) continue; // Pominięcie, jeśli seria jest nieprawidłowa (nie powinno się zdarzyć)
103
104 // Dodanie punktu (bieżący indeks próbki, wartość)
105 series->append(currentSampleIndex, axisValuesToAdd[i]);
106
107 // Usunięcie najstarszego punktu, jeśli przekroczono limit próbek
108 if (series->count() > defaultSampleCount) {
109 series->remove(0); // Usuwa punkt o indeksie 0
110 }
111 }
112
113 // Aktualizacja zakresu osi X - rzadziej, dla wydajności
114 const int xAxisUpdateFrequency = 10; // Aktualizuj co 10 próbek
115 bool isFirstSampleOrUpdateNeeded = (currentSampleIndex == 0 || currentSampleIndex % xAxisUpdateFrequency == 0);
116
117 if (isFirstSampleOrUpdateNeeded) {
118 // Bezpieczne rzutowanie na QValueAxis
119 if (auto *xAxis = qobject_cast<QValueAxis*>(chartPtr->axisX())) {
120 qint64 lastIndex = currentSampleIndex;
121 // Obliczenie minimalnego indeksu X do pokazania
122 // Jeśli mamy więcej niż defaultSampleCount punktów, pokaż ostatnie defaultSampleCount
123 qint64 minX = (lastIndex >= defaultSampleCount) ? (lastIndex - defaultSampleCount + 1) : 0;
124 qint64 maxX = lastIndex; // Maksymalny indeks X to bieżący indeks
125
126 // Korekta na wypadek, gdyby minX > maxX (np. przy bardzo małej liczbie próbek lub błędach)
127 if (minX > maxX) {
128 minX = qMax(0LL, maxX - defaultSampleCount + 1); // Upewnij się, że minX nie jest ujemne
129 }
130 // Specjalny przypadek dla samego początku (gdy nie ma jeszcze danych lub tylko 1 punkt)
131 // lub gdy defaultSampleCount jest małe
132 if (lastIndex < 1 && defaultSampleCount > 0) {
133 minX = 0;
134 maxX = defaultSampleCount -1; // Pokaż cały początkowy zakres
135 } else if (lastIndex < defaultSampleCount) {
136 minX = 0; // Pokaż od początku, dopóki nie zapełnimy okna
137 maxX = qMax(lastIndex, static_cast<qint64>(defaultSampleCount - 1)); // Rozszerzaj maxX do pełnego okna
138 }
139
140 // Ustawienie nowego zakresu osi X
141 xAxis->setRange(minX, maxX);
142 }
143 }
144
145 // Inkrementacja indeksu próbki dla następnego wywołania
146 currentSampleIndex++;
147}

149// Implementacja metody ustawiania liczby próbek
150void SensorGraph::setSampleCount(const int &sampleCount) {
151 // Ustawienie nowej liczby próbek, zapewniając minimalną wartość 10
152 defaultSampleCount = qMax(10, sampleCount);
153 QChart *chartPtr = this->chart();
154
155 // Dostosowanie liczby punktów w każdej serii do nowego limitu
156 for (QLineSeries *series : seriesList) {
157 if (series) {
158 // Usuwanie najstarszych punktów, dopóki liczba punktów nie będzie zgodna z nowym limitem
159 while (series->count() > defaultSampleCount) {
160 series->remove(0);
161 }
162 }
163 }
164
165 // Aktualizacja zakresu osi X, aby pasował do nowej liczby próbek
166 if (chartPtr) {
167 if (auto *xAxis = qobject_cast<QValueAxis*>(chartPtr->axisX())) {
168 qint64 minX = 0;
169 qint64 maxX = 0;
170
171 // Sprawdzenie, czy są jakieś punkty w pierwszej serii, aby określić bieżący zakres
172 if (!seriesList.isEmpty() && seriesList.first() && seriesList.first()->count() > 0) {
173 const auto &points = seriesList.first()->pointsVector(); // Użyj pointsVector dla wydajności
174 if (!points.isEmpty()) {
175 // Pobierz X pierwszego i ostatniego punktu *aktualnie* w serii
176 minX = static_cast<qint64>(points.first().x());
177 maxX = static_cast<qint64>(points.last().x());
178 } else {
179 // Jeśli seria jest pusta po usunięciu, ustaw domyślny zakres
180 maxX = (defaultSampleCount > 0) ? (defaultSampleCount - 1) : 0;
181 // currentSampleIndex resetujemy tylko jeśli nie ma danych
182 // Lepiej nie resetować currentSampleIndex tutaj, aby kontynuować numerację
183 }
184 } else {
185 // Jeśli nie ma żadnych serii lub są puste, ustaw zakres domyślny [0, count-1]
186 maxX = (defaultSampleCount > 0) ? (defaultSampleCount - 1) : 0;
187 // W tym przypadku reset indeksu może mieć sens, jeśli zaczynamy od zera
188 // currentSampleIndex = 0; // Rozważ, czy to jest pożądane zachowanie
189 }
190
191 // Ustawienie zakresu osi X, upewniając się, że minX <= maxX
192 if (maxX >= minX) {
193 xAxis->setRange(minX, maxX);
194 } else {
195 // Sytuacja awaryjna: ustaw domyślny zakres
196 xAxis->setRange(0, defaultSampleCount > 0 ? defaultSampleCount - 1 : 0);
197 }
198 }
199 }
200}
