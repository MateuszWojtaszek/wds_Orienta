80void SensorGraph::addData(const QVector<int> &axisValuesToAdd) {
81 QChart *chartPtr = this->chart();
82 if (axisValuesToAdd.size() != 3 || seriesList.size() != 3 || !chartPtr) {
83 return;
84 }
85
86 for (int i = 0; i < 3; ++i) {
87 QLineSeries *series = seriesList.at(i);
88 if (!series) continue;
89
90 series->append(currentSampleIndex, axisValuesToAdd[i]);
91
92 if (series->count() > defaultSampleCount) {
93 series->remove(0);
94 }
95 }
96
97 const int xAxisUpdateFrequency = 10;
98 bool isFirstSampleOrUpdateNeeded = (currentSampleIndex == 0 || currentSampleIndex % xAxisUpdateFrequency == 0);
99
100 if (isFirstSampleOrUpdateNeeded) {
101 if (auto *xAxis = qobject_cast<QValueAxis*>(chartPtr->axisX())) {
102 qint64 lastIndex = currentSampleIndex;
103 qint64 minX = (lastIndex >= defaultSampleCount) ? (lastIndex - defaultSampleCount + 1) : 0;
104 qint64 maxX = lastIndex;
105
106 if (minX > maxX) {
107 minX = qMax(0LL, maxX - defaultSampleCount + 1);
108 }
109 if (lastIndex < 1 && defaultSampleCount > 0) {
110 minX = 0;
111 maxX = defaultSampleCount -1;
112 } else if (lastIndex < defaultSampleCount) {
113 minX = 0;
114 maxX = qMax(lastIndex, static_cast<qint64>(defaultSampleCount - 1));
115 }
116 xAxis->setRange(minX, maxX);
117 }
118 }
119 currentSampleIndex++;
120}

122void SensorGraph::setSampleCount(int sampleCount) { // Zmieniono const int& na int
123 defaultSampleCount = qMax(10, sampleCount);
124 QChart *chartPtr = this->chart();
125
126 for (QLineSeries *series : seriesList) {
127 if (series) {
128 while (series->count() > defaultSampleCount) {
129 series->remove(0);
130 }
131 }
132 }
133
134 if (chartPtr) {
135 if (auto *xAxis = qobject_cast<QValueAxis*>(chartPtr->axisX())) {
136 qint64 minX = 0;
137 qint64 maxX = 0;
138
139 if (!seriesList.isEmpty() && seriesList.first() && seriesList.first()->count() > 0) {
140 const auto &points = seriesList.first()->pointsVector();
141 if (!points.isEmpty()) {
142 minX = static_cast<qint64>(points.first().x());
143 maxX = static_cast<qint64>(points.last().x());
144 } else {
145 maxX = (defaultSampleCount > 0) ? (defaultSampleCount - 1) : 0;
146 }
147 } else {
148 maxX = (defaultSampleCount > 0) ? (defaultSampleCount - 1) : 0;
149 }
150
151 if (maxX >= minX) {
152 xAxis->setRange(minX, maxX);
153 } else {
154 xAxis->setRange(0, defaultSampleCount > 0 ? defaultSampleCount - 1 : 0);
155 }
156 }
157 }
158}
