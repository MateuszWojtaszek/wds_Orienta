358void ImuDataHandler::setupModelLoader3D(Qt3DCore::QEntity *rootEntity) {
359 auto *modelEntity = new Qt3DCore::QEntity(rootEntity); // Encja dla modelu
360 auto *loader = new Qt3DRender::QSceneLoader(modelEntity); // Komponent do ładowania sceny/modelu
361
362 // Połączenie sygnału statusChanged, aby przetworzyć załadowany model
363 connect(loader, &Qt3DRender::QSceneLoader::statusChanged, this,
364 [this, loader, modelEntity](Qt3DRender::QSceneLoader::Status status) {
365 if (status == Qt3DRender::QSceneLoader::Ready) {
366 const auto &childEntities = loader->entities(); // Pobranie encji ze sceny
367 if (!childEntities.isEmpty()) {
368 Qt3DCore::QEntity *sceneRootEntity = childEntities.first();
369 if (!sceneRootEntity) return;
370
371 // Utworzenie i przypisanie transformacji do głównej encji modelu,
372 // jeśli jeszcze nie została przypisana.
373 if (!this->boardTransform) {
374 this->boardTransform = new Qt3DCore::QTransform();
375 }
376 // Sprawdzenie, czy komponent transformacji już istnieje
377 bool alreadyHasTransform = false;
378 for(Qt3DCore::QComponent *comp : sceneRootEntity->components()){
379 if(dynamic_cast<Qt3DCore::QTransform*>(comp) == this->boardTransform){
380 alreadyHasTransform = true;
381 break;
382 }
383 }
384 if(!alreadyHasTransform){
385 // Jeśli model ma już swoją transformację, można ją pobrać
386 // Qt3DCore::QTransform *existingTransform = sceneRootEntity->property("transform").value<Qt3DCore::QTransform*>();
387 // if (existingTransform) { this->boardTransform = existingTransform; }
388 // else { sceneRootEntity->addComponent(this->boardTransform); }
389 // Bezpieczniej jest dodać naszą, jeśli chcemy nią sterować z zewnątrz.
390 // Jeśli chcemy zachować oryginalną transformację modelu i na niej operować,
391 // trzeba by ją znaleźć i przypisać do boardTransform.
392 // Na razie zakładamy, że dodajemy nową lub przejmujemy kontrolę.
393 sceneRootEntity->addComponent(this->boardTransform);
394 }
395 // Przeskalowanie i ewentualne początkowe obrócenie modelu, jeśli potrzeba
396 // this->boardTransform->setScale3D(QVector3D(0.1f, 0.1f, 0.1f)); // Przykład skalowania
397 // this->boardTransform->setRotationX(90); // Przykład obrotu
398 } else {
399 qWarning() << "SceneLoader: No entities found after loading model from" << loader->source();
400 }
401 } else if (status == Qt3DRender::QSceneLoader::Error) {
402 qWarning() << "Failed to load model:" << loader->source();
403 }
404 });
405
406 // Ustawienie źródła modelu (ścieżka do pliku .dae, .gltf, .obj itp.)
407 // PAMIĘTAJ o zmianie ścieżki na poprawną dla Twojego systemu!
408 loader->setSource(QUrl::fromLocalFile("/Users/mateuszwojtaszek/projekty/wds_Orienta/ESP32.dae"));
409 // loader->setSource(QUrl("qrc:/models/ESP32.dae")); // Jeśli używasz zasobów Qt
410
411 modelEntity->addComponent(loader); // Dodanie komponentu ładowarki do encji modelu
412}
