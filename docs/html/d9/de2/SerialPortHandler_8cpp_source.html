106void SerialPortHandler::readData() {
107 // Sprawdzenie, czy port jest poprawnie zainicjalizowany, otwarty i czy można z niego czytać
108 if (!serial || !serial->isOpen() || !serial->isReadable()) {
109 return; // Wyjście z funkcji, jeśli warunki nie są spełnione
110 }
111
112 try {
113 // Sprawdzenie, czy są dostępne dane do odczytu
114 if (serial->bytesAvailable() > 0) {
115 buffer.append(serial->readAll()); // Dodanie odczytanych danych do bufora
116 } else {
117 return; // Wyjście, jeśli nie ma danych
118 }
119 } catch(...) {
120 // Obsługa nieoczekiwanego wyjątku podczas odczytu
121 qWarning() << "Exception while reading serial data.";
122 buffer.clear(); // Wyczyszczenie bufora w razie błędu
123 return;
124 }
125
126 // Przetwarzanie bufora w poszukiwaniu kompletnych linii (zakończonych '\n')
127 while (buffer.contains('\n')) {
128 int index = buffer.indexOf('\n'); // Znalezienie pozycji znaku nowej linii
129 QByteArray line = buffer.left(index).trimmed(); // Wyodrębnienie linii i usunięcie białych znaków z początku/końca
130 buffer.remove(0, index + 1); // Usunięcie przetworzonej linii z bufora
131
132 // Pominięcie pustych linii
133 if (line.isEmpty()) {
134 continue;
135 }
136
137 // Podział linii na części względem przecinka
138 QList<QByteArray> values = line.split(',');
139
140 // Sprawdzenie, czy liczba wartości zgadza się z oczekiwaną
141 if (values.size() == EXPECTED_VALUE_COUNT_SERIAL) {
142 QVector<float> parsedValues; // Wektor na sparsowane wartości
143 parsedValues.reserve(EXPECTED_VALUE_COUNT_SERIAL); // Rezerwacja pamięci
144 bool conversionOk = true; // Flaga poprawności konwersji
145
146 // Iteracja przez podzielone wartości i próba konwersji na float
147 for (const QByteArray &val : values) {
148 bool ok;
149 float floatVal = val.toFloat(&ok); // Konwersja na float
150 if (!ok) {
151 // Zgłoszenie ostrzeżenia w przypadku błędu konwersji
152 qWarning() << "Failed to convert value to float:" << val << "in line:" << line;
153 conversionOk = false; // Ustawienie flagi błędu
154 break; // Przerwanie pętli dla tej linii
155 }
156 parsedValues.append(floatVal); // Dodanie sparsowanej wartości do wektora
157 }
158
159 // Jeśli wszystkie konwersje się powiodły, emituj sygnał
160 if (conversionOk) {
161 emit newDataReceived(parsedValues);
162 }
163
164 } else {
165 // Zgłoszenie ostrzeżenia o nieprawidłowej liczbie wartości w linii
166 qWarning() << "Received line with incorrect value count (" << values.size()
167 << ", expected" << EXPECTED_VALUE_COUNT_SERIAL << "):" << line;
168 }
169 }
170}
