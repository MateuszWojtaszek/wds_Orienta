30int main(int argc, char *argv[]) {
31 QCoreApplication a(argc, argv); // Wymagane dla mechanizmu sygnałów/slotów Qt
32
33 SerialPortHandler portHandler;
34
35 // Połącz sygnały ze slotami
36 QObject::connect(&portHandler, &SerialPortHandler::newDataReceived, &onNewData);
37 QObject::connect(&portHandler, &SerialPortHandler::errorOccurred, &onError);
38
39 // Skonfiguruj i otwórz port
40 // Zastąp "COM3" lub "/dev/ttyS0" rzeczywistą nazwą portu
41 QString portName = "COM3"; // Lub "/dev/ttyUSB0", itp.
42 qint32 baudRate = 115200;
43
44 if (portHandler.openPort(portName, baudRate)) {
45 qInfo() << "Pomyślnie otwarto port" << portName;
46 // Port jest teraz otwarty, a readData() będzie wywoływane asynchronicznie,
47 // gdy nadejdą dane.
48 } else {
49 qWarning() << "Nie udało się otworzyć portu" << portName << ". Błąd:" << portHandler.getLastError();
50 // Aplikacja może zakończyć działanie lub spróbować innych portów, jeśli otwarcie się nie powiedzie
51 return -1;
52 }
53
54 qInfo() << "Aplikacja uruchomiona. Oczekiwanie na dane z portu szeregowego...";
55
56 // Uruchom pętlę zdarzeń Qt, aby umożliwić przetwarzanie sygnałów/slotów
57 // i utrzymać działanie aplikacji.
58 int execResult = a.exec();
59
60 // Sprzątanie: zamknij port po zakończeniu (chociaż destruktor również to obsługuje)
61 portHandler.closePort();
62 qInfo() << "Aplikacja zakończona.";
